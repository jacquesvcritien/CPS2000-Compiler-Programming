// Function definition for Power
ff Pow(x : float , n : int ) : auto
{
    let y : float = 1.0 ; // Declare y and set it to 1.0
    if( n>0 )
    {
        for (; n>0 ; n=n-1)
        {
            y = y*x; //Assignment y = y*x;
        }
    }
    else
    {
        for (; n<0 ; n=n+1)
        {
            y = y/x; //Assignment y = y/x;
        }
    }
    return y ; // return y as the result
}

print Pow;
